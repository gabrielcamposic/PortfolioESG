# --- Optimized Portfolio Script Parameters for PortfolioESG ---
# Script C_OptimizedPortfolio.py: combines ideal portfolio (A) with holdings (B)

# --- Files and paths ---
OPTIMIZED_RESULTS_FILE = ~/PortfolioESG/data/results/optimized_portfolio_history.csv

# --- Score Weights (must sum to 1.0) ---
# Pesos para score do portfolio otimizado

TRANSACTION_COST_FIXED_PCT = 0.03

# --- Expected Return Window ---
# Janela temporal para cálculo de expected return (dias úteis)

# --- Optimization Settings ---
# Número de portfolios candidatos a gerar entre holdings e ideal
NUM_CANDIDATE_PORTFOLIOS = 100

# Mínimo de overlap (%) entre holdings e recomendação para considerar transição
MIN_OVERLAP_THRESHOLD = 0.0

# Máximo de transações permitidas em uma recomendação (0 = sem limite)
